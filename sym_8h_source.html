21/// A Sym%bol just wraps a pointer to Sym::String, so pass Sym itself around as value.
22/// Sym is compatible with:
23/// * recommended: `std::string_view` (via Sym::view)
24/// * null-terminated C-strings (via Sym::c_str)
25///
26/// This means that retrieving a `std::string_view` or a null-terminated C-string is basically free.
27/// You can also obtain a `std::string` (via Sym::str), but this involves a copy.
28/// With the exception of the empty string, you should only create Sym%bols via SymPool::sym.
29/// This in turn will toss all Sym%bols into a big hash set.
30/// This makes Sym::operator== and Sym::operator!= an O(1) operation.
31/// The empty string is internally handled as `nullptr`.
32/// Thus, you can create a Sym%bol representing an empty string without having access to the SymPool.
33/// @note The empty `std::string`/`std::string_view`, `nullptr`, and `"\0"` are all identified as Sym::Sym().

43 struct Equal {
44 bool operator()(const String* s1, const String* s2) const {
45 bool res = s1->size == s2->size;
46 for (size_t i = 0, e = s1->size; res && i != e; ++i) res &= s1->chars[i] == s2->chars[i];
47 return res;
48 }

189 /// @name sym
190 ///@{
191 Sym sym(std::string_view s) {
192 if (s.empty()) return Sym();
193 auto state = strings_.state();
194 auto ptr = (String*)strings_.allocate(sizeof(String) + s.size() + 1 /*'\0'*/);
195 new (ptr) String(s.size());
196 *std::copy(s.begin(), s.end(), ptr->chars) = '\0';
197 auto [i, ins] = pool_.emplace(ptr);
198 if (ins) return Sym(ptr);
199 strings_.deallocate(state);
200 return Sym(*i);
201 }
202 Sym sym(const std::string& s) { return sym((std::string_view)s); }
203 /// @p s is a null-terminated C-string.
204 Sym sym(const char* s) { return s == nullptr || *s == '\0' ? Sym() : sym(std::string_view(s, strlen(s))); }
205 // TODO we can try to fit s in current page and hence eliminate the explicit use of strlen
206 ///@}
